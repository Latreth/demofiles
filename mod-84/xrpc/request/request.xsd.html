<?xml version="1.0" encoding="utf-8"?>
<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema">

 <!-- Elément racine d'une requête XML-RPC -->

 <xs:element name="methodCall">
  <xs:complexType>
   <xs:sequence>
    <xs:element name="methodName" type="T_NAME"></xs:element>
    <xs:element name="params" minOccurs="0" maxOccurs="1" type="T_PARAMS"></xs:element>
   </xs:sequence>
  </xs:complexType>  
 </xs:element>


 <!-- Nom d'une méthode -->

 <xs:simpleType name="T_NAME">
  <xs:restriction base="xs:string">
   <xs:pattern value="[A-Za-z0-9/.:_]+"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>


 <!-- Liste des paramètres d'une méthode -->

 <xs:complexType name="T_PARAMS">
  <xs:sequence>
   <xs:element name="param" minOccurs="0" maxOccurs="unbounded" type="T_PARAM"></xs:element>
  </xs:sequence>
 </xs:complexType>


 <!-- Paramètre d'une méthode -->

 <xs:complexType name="T_PARAM">
  <xs:choice>
   <xs:element name="value" type="T_VALUE"></xs:element>
  </xs:choice>
 </xs:complexType>

</xs:schema>
