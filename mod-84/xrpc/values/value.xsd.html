Texte source du fichier : value.xsd

<?xml version="1.0" encoding="ISO-8859-1"?>
<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema">

 <!-- Valeurs XML-RPC -->

 <!-- N.B. Pour autoriser une valeur chaîne sans la balise "string" -->
 <!--      il faut que ce type soit à contenu mixte (mixed="true")  -->
 <!--      ce qui a l'inconvénient de valider des docs non valides  -->

 <xs:complexType name="T_VALUE" mixed="true">
  <xs:choice>
   <xs:element name="int" type="T_INT"></xs:element>
   <xs:element name="i4" type="T_INT"></xs:element>
   <xs:element name="boolean" type="T_BOOLEAN"></xs:element>
   <xs:element name="double" type="T_DOUBLE"></xs:element>
   <xs:element name="dateTime.iso8601" type="T_DATE"></xs:element>
   <xs:element name="string" type="T_STRING"></xs:element>
   <xs:element name="base64" type="T_BASE64"></xs:element>
   <xs:element name="array" type="T_ARRAY"></xs:element>
   <xs:element name="struct" type="T_STRUCT"></xs:element>
  </xs:choice>
 </xs:complexType>


 <!-- Valeur Booléenne -->

 <xs:simpleType name="T_BOOLEAN">
  <xs:restriction base="xs:string">
   <xs:pattern value="0|1"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>


 <!-- Valeur Entière -->

 <xs:simpleType name="T_INT">
  <xs:restriction base="xs:string">
   <xs:pattern value="[+\-]?(0+|0*([1-9][0-9]{0,8}|1[0-9]{9}|2(0[0-9]{8}|1([0-3][0-9]{7}|4([0-6][0-9]{6}|7([0-3][0-9]{5}|4([0-7][0-9]{4}|8([0-2][0-9]{3}|3([0-5][0-9]{2}|6([0-3][0-9]|4[0-7]))))))))))|-0*2147483648"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>


 <!-- Valeur Décimale -->

 <xs:simpleType name="T_DOUBLE">
  <xs:restriction base="xs:string">
   <xs:pattern value="[+\-]?[0-9]+\.[0-9]+"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>


 <!-- Valeur Date -->

 <xs:simpleType name="T_DATE">
  <xs:restriction base="xs:string">
   <xs:pattern value="([0-9]{4}((0(1|3|5|7|8)|1(0|2))(0[1-9]|[1-2][0-9]|3[0-1])|(0(4|6|8)|10)(0[1-9]|[1-2][0-9]|30)|02(0[1-9]|1[0-9]|2[0-8]))|(([0-9]{2}(0(4|8)|(1|3|5|7|9)(2|4|6)|(2|4|6|8)(0|4|8))|[0-9]000)0229))T(((0|1)[0-9])|(2[0-3])):[0-5][0-9]:[0-5][0-9]"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>


 <!-- Valeur Chaîne de caractères -->

 <xs:simpleType name="T_STRING">
  <xs:restriction base="xs:string">
   <xs:pattern value="(&#x9;|&#xD;|&#xA;|[&#x20;-&#x7F;])*"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>


 <!-- Valeur codée base 64 -->

 <xs:simpleType name="T_BASE64">
  <xs:restriction base="xs:base64Binary"></xs:restriction>
 </xs:simpleType>


 <!-- Valeur Tableau -->

 <xs:complexType name="T_ARRAY">
  <xs:sequence>
   <xs:element name="data" type="T_ARRAY_DATA"></xs:element>
  </xs:sequence>
 </xs:complexType>

 <xs:complexType name="T_ARRAY_DATA">
  <xs:sequence>
   <xs:element name="value" type="T_VALUE" minOccurs="0" maxOccurs="unbounded"></xs:element>
  </xs:sequence>
 </xs:complexType>


 <!-- Valeur Structure -->

 <xs:complexType name="T_STRUCT">
  <xs:sequence>
   <xs:element name="member" maxOccurs="unbounded" type="T_STRUCT_MEMBER"></xs:element>
  </xs:sequence>
 </xs:complexType>

 <xs:complexType name="T_STRUCT_MEMBER">
  <xs:sequence>
   <xs:element name="name" type="T_NAME"></xs:element>
   <xs:element name="value" type="T_VALUE"></xs:element>
  </xs:sequence>
 </xs:complexType>


</xs:schema>
