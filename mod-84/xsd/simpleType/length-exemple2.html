<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema">

 <xs:element name="http">
  <xs:complexType>
   <xs:sequence>
    <xs:element ref="request"></xs:element>
   </xs:sequence>
  </xs:complexType>
 </xs:element>

 <xs:element name="request">
  <xs:complexType>
   <xs:sequence>
    <xs:element name="Host" type="T_HOST"></xs:element>
    <xs:element name="User-Agent" type="T_USER_AGENT"></xs:element>
    <xs:element name="Authorization" type="T_AUTHORIZATION" minOccurs="0"></xs:element>
   </xs:sequence>
   <xs:attribute name="method" type="T_HTTP_METHOD" use="required"></xs:attribute>
   <xs:attribute name="URL" type="T_URL" use="required"></xs:attribute>
   <xs:attribute name="version" type="T_VERSION"></xs:attribute>
  </xs:complexType>
 </xs:element>

 <xs:simpleType name="T_HTTP_METHOD">
  <xs:restriction base="xs:NCName">
   <xs:pattern value="GET|HEAD"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>
 
 <xs:simpleType name="T_URL">
  <xs:restriction base="xs:anyURI">
   <xs:pattern value="/.*"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>
 
 <xs:simpleType name="T_VERSION">
  <xs:restriction base="xs:token">
   <xs:pattern value="HTTP/1.[0-1]"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>
 
 <xs:simpleType name="T_HOST">
  <xs:restriction base="xs:token">
   <xs:pattern value="[A-Za-z0-9_\.\-]*"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>
 
 <xs:simpleType name="T_USER_AGENT">
  <xs:restriction base="xs:token"></xs:restriction>
 </xs:simpleType>
 
 <xs:complexType name="T_AUTHORIZATION">
  <xs:simpleContent>
   <xs:extension base="T_BASIC_PASSWORD">
    <xs:attribute name="method" type="T_AUTHENTICATION_METHOD"></xs:attribute>
   </xs:extension>
  </xs:simpleContent>
 </xs:complexType>
 
 <xs:simpleType name="T_AUTHENTICATION_METHOD">
  <xs:restriction base="xs:NCName">
   <xs:pattern value="Basic"></xs:pattern>
  </xs:restriction>
 </xs:simpleType>

 <xs:simpleType name="T_BASIC_PASSWORD">
  <xs:restriction base="xs:base64Binary">
   <xs:length value="13"></xs:length>
  </xs:restriction>
 </xs:simpleType>

</xs:schema>
