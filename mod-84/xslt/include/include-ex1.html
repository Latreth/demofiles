Feuille de style XSLT

[fermer]
<?xml version="1.0" encoding="utf-8"?>
<xsl:stylesheet version="1.0" xmlns:xsl="http://www.w3.org/1999/XSL/Transform">

 <xsl:output method="html" encoding="utf-8" doctype-public="-//W3C//DTD XHTML 1.0 Transitional//EN" doctype-system="http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd"></xsl:output>

 <xsl:include href="include-ex1i.xsl"></xsl:include>

 <xsl:template match="étudiant">
  <tr>
   <xsl:apply-templates></xsl:apply-templates>
  </tr>
 </xsl:template>

 <xsl:template match="nom | prénom">
  <td>
   <xsl:apply-templates></xsl:apply-templates>
  </td>
 </xsl:template>

</xsl:stylesheet>
