 <script type="text/javascript">
    var timevalue = 0;
    var timer_increment = 50;
    var max_time = 3000;
    var text_element;

    function ButtonDown(evt,bname) {
	var button;
	button=evt.target.ownerDocument.getElementById(bname);
      	button.setAttribute("filter", "url(#ButtonIn)");
    }

    function ButtonUp(evt,bname) {
	var button;
	button=evt.target.ownerDocument.getElementById(bname);
      	button.setAttribute("filter", "url(#ButtonOut)");
    }

    function StartAnimation(evt,elt) {
      text_element = evt.target.ownerDocument.getElementById(elt);
      text_element.setAttribute("visibility", "visible");
      timevalue = 0;
      ShowAndGrowElement();
    }

    function ShowAndGrowElement() {
      timevalue = timevalue + timer_increment;
      if (timevalue > max_time)
        return;

      // Scale the text string gradually until it is 20 times larger
      scalefactor = (timevalue * 20.) / max_time;
      scale = "scale(" + scalefactor + ")";
      transx = 700. - 450. * timevalue / max_time;
      transy = 80. + 70. * timevalue / max_time;

      text_element.setAttribute("transform", "translate(" + transx + " " + transy + ") scale(" + scalefactor + ")");

      // Call ShowAndGrowElement again timer_increment milliseconds later.
      setTimeout("ShowAndGrowElement()", timer_increment)
    }

 </script>
